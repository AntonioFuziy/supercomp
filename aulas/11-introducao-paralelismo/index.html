    import subprocess
import time
def roda_com_entrada(exec, arq):
    with open(arq) as f:
        start = time.perf_counter()
        proc = subprocess.run([exec], input=f.read(), text=True, capture_output=True)
        end = time.perf_counter()

        return proc.stdout, end - start
